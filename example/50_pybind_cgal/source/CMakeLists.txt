cmake_minimum_required( VERSION 3.4 )
project( mytest LANGUAGES CXX )
set( CMAKE_VERBOSE_MAKEFILE ON )
set( CMAKE_CXX_STANDARD 14 )
find_package( Python COMPONENTS Interpreter Development REQUIRED )
find_package( pybind11 REQUIRED )
find_package( CGAL QUIET COMPONENTS Core )
set( SRCS vis_00_base.cpp )
add_executable( mytest ${SRCS} )
target_link_libraries( mytest PRIVATE pybind11::embed gmp )
pybind11_add_module( cgal_mesher vis_01_mesher_module.cpp )
target_link_libraries( cgal_mesher PRIVATE gmp )
target_precompile_headers( cgal_mesher PRIVATE vis_01_mesher_module.hpp )
