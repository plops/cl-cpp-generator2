cmake_minimum_required( VERSION 3.16 FATAL_ERROR )
project( mytest )
set( CMAKE_VERBOSE_MAKEFILE ON )
set( SRCS /home/martin/stage/cl-cpp-generator2/example/99_qt_pch/source00/main.cpp )
add_executable( mytest ${SRCS} )
set_property( TARGET mytest PROPERTY CXX_STANDARD 20 )
find_package( PkgConfig REQUIRED )
pkg_check_modules( spdlog REQUIRED spdlog )
pkg_check_modules( Qt5Gui REQUIRED Qt5Gui )
pkg_check_modules( Qt5Core REQUIRED Qt5Core )
pkg_check_modules( Qt5Widgets REQUIRED Qt5Widgets )
target_include_directories( mytest PUBLIC ${spdlog_INCLUDE_DIRS} ${Qt5Gui_INCLUDE_DIRS} ${Qt5Core_INCLUDE_DIRS} ${Qt5Widgets_INCLUDE_DIRS} )
target_compile_options( mytest PUBLIC ${spdlog_CFLAGS_OTHER} ${Qt5Gui_CFLAGS_OTHER} ${Qt5Core_CFLAGS_OTHER} ${Qt5Widgets_CFLAGS_OTHER} )
target_link_libraries( mytest PRIVATE ${spdlog_LIBRARIES} ${Qt5Gui_LIBRARIES} ${Qt5Core_LIBRARIES} ${Qt5Widgets_LIBRARIES} )
target_precompile_headers( mytest PRIVATE fatheader.hpp )
