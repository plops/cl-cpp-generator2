cmake_minimum_required( VERSION 3.19 )
project( mytest LANGUAGES CXX )
find_package( mdspan REQUIRED )
set( CMAKE_CXX_COMPILER clang++ )
set( CMAKE_VERBOSE_MAKEFILE ON )
set (CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -ggdb -O0 -std=gnu++2b  -Wlogical-op -Wnoexcept  -Wstrict-null-sentinel  -Wsign-promo-Wstrict-overflow=5   ")
set (CMAKE_LINKER_FLAGS_DEBUG "${CMAKE_LINKER_FLAGS_DEBUG} -ggdb -O0 -std=gnu++2b -Wlogical-op -Wnoexcept  -Wstrict-null-sentinel  -Wsign-promo-Wstrict-overflow=5   ")
set( SRCS  )
add_executable( mytest ${SRCS} )
